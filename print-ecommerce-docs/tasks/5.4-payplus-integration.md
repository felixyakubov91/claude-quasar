# Task 5.4: PayPlus Integration

## Goal
Integrate PayPlus payment gateway.

## Instructions

### Create Payment Page
Create `frontend/src/pages/checkout/PaymentPage.vue`:

```vue
<template>
  <q-page class="q-pa-md">
    <div class="row justify-center">
      <div class="col-12 col-md-6">
        <q-card>
          <q-card-section>
            <div class="text-h6">{{ $t('checkout.payment') }}</div>
            <p class="text-grey-7">{{ $t('checkout.paymentDescription') }}</p>
          </q-card-section>

          <q-card-section>
            <div class="text-center q-py-xl" v-if="loading">
              <q-spinner size="50px" color="primary" />
              <p class="q-mt-md">{{ $t('checkout.redirecting') }}</p>
            </div>
          </q-card-section>
        </q-card>
      </div>
    </div>
  </q-page>
</template>

<script setup lang="ts">
import { ref, onMounted } from 'vue';
import { useRoute, useRouter } from 'vue-router';
import { useQuasar } from 'quasar';
import { useCartStore } from 'src/stores/cart.store';
import { useAuth } from 'src/composables/useAuth';
import { strapiService } from 'src/services/strapi.service';
import { paymentService } from 'src/services/payment.service';

const $q = useQuasar();
const route = useRoute();
const router = useRouter();
const cartStore = useCartStore();
const { user, session } = useAuth();

const loading = ref(true);

onMounted(async () => {
  // Check for error from PayPlus redirect
  if (route.query.error) {
    $q.notify({ type: 'negative', message: 'Payment failed' });
    loading.value = false;
    return;
  }

  try {
    const checkoutData = JSON.parse(sessionStorage.getItem('checkoutData') || '{}');
    
    // Create order in Strapi
    const order = await strapiService.createOrder({
      customer: user.value?.id,
      items: cartStore.items.map(item => ({
        product: item.productId,
        quantity: item.quantity,
        unit_price: item.unitPrice,
        selected_options: item.selectedOptions,
      })),
      subtotal: cartStore.subtotal,
      shipping_address: checkoutData.shippingAddress,
      shipping_method: checkoutData.shippingMethod?.name,
      shipping_cost: checkoutData.shippingMethod?.price || 0,
      status: 'pending',
      payment_status: 'pending',
    }, session.value?.access_token);

    // Initialize PayPlus payment
    const paymentUrl = await paymentService.createPayment({
      orderId: order.data.id,
      amount: cartStore.subtotal + (checkoutData.shippingMethod?.price || 0),
      customerEmail: user.value?.email,
      customerName: checkoutData.shippingAddress?.full_name,
    });

    // Redirect to PayPlus
    window.location.href = paymentUrl;
  } catch (error) {
    $q.notify({ type: 'negative', message: 'Failed to process payment' });
    loading.value = false;
  }
});
</script>
```

### Create Payment Service
Create `frontend/src/services/payment.service.ts`:

```typescript
import axios from 'axios';
import { env } from 'src/utils/env';

class PaymentService {
  private baseUrl = env.payPlusMode === 'live' 
    ? 'https://api.payplus.co.il'
    : 'https://sandbox.payplus.co.il';

  async createPayment(data: {
    orderId: number;
    amount: number;
    customerEmail: string;
    customerName: string;
  }): Promise<string> {
    const response = await axios.post(`${env.strapiUrl}/api/payments/create`, data);
    return response.data.paymentUrl;
  }
}

export const paymentService = new PaymentService();
```

### Create Strapi Payment Controller
Create `backend/src/api/payment/controllers/payment.js`:

```javascript
const axios = require('axios');

module.exports = {
  async create(ctx) {
    const { orderId, amount, customerEmail, customerName } = ctx.request.body;

    const response = await axios.post(`${process.env.PAYPLUS_API_URL}/Transactions/GenerateLink`, {
      payment_page_uid: process.env.PAYPLUS_PAGE_UID,
      charge_method: 1,
      amount,
      currency_code: 'ILS',
      sendEmailApproval: true,
      customer: {
        customer_name: customerName,
        email: customerEmail,
      },
      more_info: String(orderId),
      success_url: `${process.env.FRONTEND_URL}/checkout/confirmation?order=${orderId}`,
      failure_url: `${process.env.FRONTEND_URL}/checkout/payment?error=failed`,
    }, {
      headers: {
        Authorization: JSON.stringify({
          api_key: process.env.PAYPLUS_API_KEY,
          secret_key: process.env.PAYPLUS_SECRET_KEY,
        }),
      },
    });

    return { paymentUrl: response.data.data.payment_page_link };
  },

  async webhook(ctx) {
    const { transaction_uid, status_code, more_info } = ctx.request.body;

    if (status_code === '000') {
      await strapi.entityService.update('api::order.order', more_info, {
        data: {
          payment_status: 'paid',
          payment_reference: transaction_uid,
          paid_at: new Date(),
          status: 'confirmed',
        },
      });
    }

    return { success: true };
  },
};
```
